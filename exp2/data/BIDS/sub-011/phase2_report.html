# list data files
file_pattern <- paste("^", cSub, "_premup-three_ses-02_part-02_label-phase2_*", sep = "")
filename  <- list.files(path= data_dir, pattern= file_pattern, full.names=T)

# import data from csv file
phase2 <- read.csv(filename)
  
# Remove practice trials
#phase2 <- phase2[11:length(phase2$trial_acc),]
  
# Remove weird extra line
phase2 <- phase2[!phase2$scn_file=="",]

# Get some experiment parameters
p <- hash()
p["nIterations"] =  1
p["nTrials"] =  (length(phase2$iteration_index) / p$nIterations) - 4
p["scn_cont"] = contingencies

# Separate learning from rating trials and from encoding
learning_data <- subset(x = phase2,
    subset = trial_index<241,
    select = c(participant, trial_index, trial_cond,
               scn_file, scn_cat, scn_inst, scn_condition, fillers,
               obj_cat, obj_cat_num, obj_file, quadrant, corr_ans,
               key_resp_trial.keys, key_resp_trial.rt, trial_acc))

# Encoding
enc_data <- subset(x = phase2,
    subset = trial_index>240,
    select = c(participant, trial_index, trial_cond,
               scn_file, scn_cat, scn_inst, scn_condition, fillers,
               obj_cat, obj_cat_num, obj_file, quadrant, corr_ans,
               key_resp_trial.keys, key_resp_trial.rt, trial_acc))

# Check if slider2 recorded a response
if("slider_2.response" %in% colnames(phase2)){
rating_data <- subset(x = phase2,
    subset = !is.na(slider.response) | !is.na(slider_2.response),
    select = c(participant, 
               scn_file, scn_cat, obj_cat_top, min_resp_top, max_resp_top, 
               slider.response, slider.rt))
}else{
  rating_data <- subset(x = phase2,
    subset = !is.na(slider.response),
    select = c(participant, 
               scn_file, scn_cat, obj_cat_top, min_resp_top, max_resp_top, 
               slider.response, slider.rt))
}
rating_data$iteration_index <- c(rep(1,6),rep(2,6))
colnames(enc_data)[colnames(enc_data)=="trial_index"] <- "trialN"
learning_data$trial_acc_manual[learning_data$key_resp_trial.keys == learning_data$corr_ans] <- 1

learning_data$trial_acc_manual[is.na(learning_data$key_resp_trial.keys)] <- 0


enc_data$trial_acc_manual[enc_data$key_resp_trial.keys == enc_data$corr_ans] <- 1

enc_data$trial_acc_manual[is.na(enc_data$key_resp_trial.keys)] <- 0

Compute performance indeces for refreshment

# Sort rating data
rating_data <- rating_data[with(rating_data, order(iteration_index, scn_cat)),]
final_rating <- rating_data$slider.response[rating_data$iteration_index==max(rating_data$iteration_index)]

# Get rating info
rating_data$corr_rat <- rowMeans(cbind(rating_data$min_resp_top,rating_data$max_resp_top))
# rating_data$cat_cond <- rep(NA, length(rating_data$corr_rat))

# Get cummulative accuracy split by scenes
par(mfrow=c(1,2), oma = c(2, 2, 2, 2)) # Subplots
for (cScn in seq(1,6, by=2)){
  cumm_acc <- cumsum(learning_data$trial_acc_manual[learning_data$scn_cat==cScn])
  cumm_acc2 <- cumsum(learning_data$trial_acc_manual[learning_data$scn_cat==cScn+1])
  cumm_acc <- cumm_acc / c(1:length(cumm_acc)) # Cummulative accuracy
  cumm_acc2 <- cumm_acc2 / c(1:length(cumm_acc2)) # Cummulative accuracy
 
  # Get current scene's info
  curr_scn_cond <- unique(learning_data$scn_condition[learning_data$scn_cat==cScn])
  curr_scn_cond_label <- p$scn_cont[curr_scn_cond]
  
  # Plot
  plot(cumm_acc, ylim = c(0,1), xlim = c(0,160),xaxt="n", col="pink",
       main = paste(curr_scn_cond_label, "contingency"))
  abline(h=.5, col = "red", lty = 2); abline(h=curr_scn_cond_label ,col = "green")
  abline(v=0, col = "black", lty = 4);abline(v=40, col = "black", lty = 4); abline(v=80, col = "black", lty = 4); 
  abline(v=120, col = "black", lty = 4);abline(v=160, col = "black", lty = 4);
  xtick<-seq(0, 160, by=40)
  axis(side=1, at=xtick, labels = FALSE)
  text(x=xtick,  par("usr")[3],
  labels = c("1st", "2nd", "3rd", "4th", "hard-limit"), pos = 1, xpd = TRUE)
  points(seq(1, length(cumm_acc2)), cumm_acc2, col = "blue")
  text(labels= final_rating[cScn],x= length(cumm_acc)-10,y= .1, col="pink")
  text(labels= final_rating[cScn+1],x= length(cumm_acc2)-10,y= .2, col="blue")
}
